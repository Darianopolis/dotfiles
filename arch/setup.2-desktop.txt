--------------------------------------------------------------------------------
--        Disable libinput mouse button debouncing
--------------------------------------------------------------------------------

> /etc/libinput/local-overrides.quirks
  [Never Debounce]
  MatchUdevType=mouse
  ModelBouncingKeys=1

! This requires a restart of any applicaiton that uses libinput (e.g. the compositor) to take affect
